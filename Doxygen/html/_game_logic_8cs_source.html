226  int possibleScore = this.score - (this.MainCharacter.NumberOfNewCollisionsWithRoommate * MinusScoreRoommate) - (this.FoldingMiniGame.NumberOfFailedFoldings * MinusScoreWrongArrow);

228  {
229  this.MainCharacter.NumberOfNewCollisionsWithRoommate = 0;

232  return 0;

236  this.FoldingMiniGame.NumberOfFailedFoldings = 0;
237  this.score = possibleScore;
238  return this.score;
